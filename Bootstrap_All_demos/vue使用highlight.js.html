            //读取JSON，然后把JSON转成VUE的代码
            import generator from "@/designer/codeparser/codegenerator.js";

            export default {
                name: "CodeView",
                data() {
                return {
                    code: "",
                    viewModal: false
                };
            },
            computed: {
                pageInfo() {
                    return this.$store.state.page.pageInfo;
                },
                pageDesign() {
                    return this.$store.state.page.pageDesign;
                }
            },
            methods: {
                buildCode() {
                    this.viewModal = !this.viewModal;
                    if (this.viewModal) {
                        this.code = generator(this.pageInfo, this.pageDesign);
                    }
                }
            },
            mounted() {}
            };
